\begin{verbatim}
void Sort(int[] array)
{
    for (int i=1;i<array.Length;i++)
        for (int j=0;j<i;j++)
            if (array[j]>array[j+1])
            {
                var temp=array[j];
                array[j]=array[j+1];
                array[j+1]=temp;
            }    
}
\end{verbatim}
